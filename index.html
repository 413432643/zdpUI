# 快速开始

# 安装组件库

npm i zdp-ui

# 引用组件库

// 全部引用
import '../node_modules/zdpui/dist/style.css'
import ZUI from 'zdpui'
Vue.use(ZUI)

// 按需引用
import '../node_modules/zdpui/dist/style.css'
import {Button} from 'zdpui'
Vue.use(Button)
